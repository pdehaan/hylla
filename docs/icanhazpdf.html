function getPdfFromPlos(doi) {
  var PLOS = {
    'cbi': 'compbiol',
    'med': 'medicine',
    'pat': 'pathogens',
    'one': 'one',
    'ntd': 'ntds',
    'bio': 'biology'
  };
  var match = /10\.1371\/journal\.p(pat|one|med|gen|ntd|bio|cbi)\./.exec(doi);
  var journal = match[1];
  var url = 'http://www.plos' + PLOS[journal] +
    '.org/article/fetchObject.action?uri=info:doi/' + doi +
    '&representation=PDF';
  return url;
}
